    <script src="/socket.io/socket.io.js"></script>
    <script>
        // Get room ID from URL
        const pathParts = window.location.pathname.split('/');
        const roomId = pathParts[pathParts.length - 1];
        
        // DOM Elements
        const roomIdDisplay = document.getElementById('roomIdDisplay');
        const userCount = document.getElementById('userCount');
        const statusDot = document.getElementById('statusDot');
        const statusText = document.getElementById('statusText');
        const videoUrl = document.getElementById('videoUrl');
        const loadBtn = document.getElementById('loadBtn');
        const playBtn = document.getElementById('playBtn');
        const pauseBtn = document.getElementById('pauseBtn');
        const seekTime = document.getElementById('seekTime');
        const seekBtn = document.getElementById('seekBtn');
        const videoFrame = document.getElementById('videoFrame');
        const videoPlaceholder = document.getElementById('videoPlaceholder');
        const chatMessages = document.getElementById('chatMessages');
        const chatInput = document.getElementById('chatInput');
        const sendBtn = document.getElementById('sendBtn');
        const syncStatus = document.getElementById('syncStatus');
        const connectionStatus = document.getElementById('connectionStatus');
        
        // User setup
        let username = localStorage.getItem('familySyncUsername');
        if (!username) {
            username = prompt("Enter your name for the family chat:");
            if (!username) {
                username = "Family Member";
            }
            localStorage.setItem('familySyncUsername', username);
        }
        
        // Initialize
        roomIdDisplay.textContent = roomId;
        
        // Connect to Socket.io
        const socket = io();
        
        // Connection events
        socket.on('connect', () => {
            statusDot.className = 'status-dot connected';
            statusText.textContent = 'Connected';
            
            // Join the room
            socket.emit('join-room', roomId, username);
        });
        
        socket.on('disconnect', () => {
            statusDot.className = 'status-dot';
            statusText.textContent = 'Disconnected';
        });
        
        // Room events
        socket.on('room-state', (state) => {
            console.log('Room state received:', state);
            
            // Update user count
            userCount.textContent = state.users.length;
            
            // Update video if exists
            if (state.videoUrl) {
                videoUrl.value = state.videoUrl;
                loadVideo(state.videoUrl);
                
                if (state.isPlaying) {
                    showSyncStatus('Video is playing...');
                }
            }
            
            // Update connection status
            connectionStatus.innerHTML = `
                <div style="margin-top: 15px; padding: 10px; background: rgba(255,255,255,0.1); border-radius: 5px;">
                    <strong>Connected with:</strong><br>
                    ${state.users.map(u => `â€¢ ${u}`).join('<br>')}
                </div>
            `;
        });
        
        socket.on('user-joined', (data) => {
            userCount.textContent = data.usersCount;
            addChatMessage(`${data.username} joined the room!`, 'System');
            
            connectionStatus.innerHTML += `
                <div style="margin-top: 5px; color: var(--success);">
                    <i class="fas fa-user-plus"></i> ${data.username} connected
                </div>
            `;
        });
        
        socket.on('user-left', (data) => {
            addChatMessage(`${data.username} left the room`, 'System');
        });
        
        socket.on('room-full', () => {
            alert('Room is full (max 2 users). This is your private family room.');
        });
        
        // Video control events
        socket.on('video-changed', (url) => {
            videoUrl.value = url;
            loadVideo(url);
            addChatMessage('Video changed', 'System');
        });
        
        socket.on('video-played', (time) => {
            showSyncStatus(`Playing at ${formatTime(time)}`);
            addChatMessage('Video played', 'System');
        });
        
        socket.on('video-paused', (time) => {
            showSyncStatus(`Paused at ${formatTime(time)}`);
            addChatMessage('Video paused', 'System');
        });
        
        socket.on('video-seeked', (time) => {
            showSyncStatus(`Seeked to ${formatTime(time)}`);
            addChatMessage(`Seeked to ${formatTime(time)}`, 'System');
        });
        
        // Chat events
        socket.on('receive-chat', (data) => {
            addChatMessage(data.message, data.username, data.timestamp);
        });
        
        // UI Event Handlers
        loadBtn.addEventListener('click', () => {
            const url = videoUrl.value.trim();
            if (url) {
                socket.emit('change-video', roomId, url);
                loadVideo(url);
                addChatMessage(`Loaded: ${url}`, username);
            }
        });
        
        playBtn.addEventListener('click', () => {
            const time = parseFloat(seekTime.value) || 0;
            socket.emit('play-video', roomId, time);
            showSyncStatus(`You played at ${formatTime(time)}`);
            addChatMessage('â–¶ï¸ Play', username);
        });
        
        pauseBtn.addEventListener('click', () => {
            const time = parseFloat(seekTime.value) || 0;
            socket.emit('pause-video', roomId, time);
            showSyncStatus(`You paused at ${formatTime(time)}`);
            addChatMessage('â¸ï¸ Pause', username);
        });
        
        seekBtn.addEventListener('click', () => {
            const time = parseFloat(seekTime.value);
            if (!isNaN(time)) {
                socket.emit('seek-video', roomId, time);
                showSyncStatus(`You seeked to ${formatTime(time)}`);
                addChatMessage(`â© Seek to ${formatTime(time)}`, username);
            }
        });
        
        sendBtn.addEventListener('click', sendChatMessage);
        chatInput.addEventListener('keypress', (e) => {
            if (e.key === 'Enter') sendChatMessage();
        });
        
        // Helper Functions
        function loadVideo(url) {
            // Show iframe
            videoPlaceholder.style.display = 'none';
            videoFrame.style.display = 'block';
            
            // Try to load the URL in iframe
            try {
                videoFrame.src = url;
            } catch (error) {
                videoPlaceholder.innerHTML = `
                    <div>
                        <i class="fas fa-exclamation-triangle" style="color: #f56565; font-size: 3em;"></i>
                        <p>Cannot load video directly</p>
                        <p style="font-size: 0.9em; margin-top: 10px;">
                            Open this link in a new tab:<br>
                            <a href="${url}" target="_blank" style="color: #667eea;">${url}</a>
                        </p>
                        <p style="font-size: 0.8em; margin-top: 10px; color: #888;">
                            Use the sync controls to stay in sync while watching
                        </p>
                    </div>
                `;
                videoPlaceholder.style.display = 'flex';
                videoFrame.style.display = 'none';
            }
        }
        
        function sendChatMessage() {
            const message = chatInput.value.trim();
            if (message) {
                socket.emit('send-chat', roomId, message, username);
                addChatMessage(message, username);
                chatInput.value = '';
            }
        }
        
        function addChatMessage(message, sender, timestamp = new Date().toLocaleTimeString()) {
            const messageDiv = document.createElement('div');
            messageDiv.className = 'message';
            messageDiv.innerHTML = `
                <div class="message-header">
                    <span class="message-user">${sender}</span>
                    <span class="message-time">${timestamp}</span>
                </div>
                ${message}
            `;
            chatMessages.appendChild(messageDiv);
            chatMessages.scrollTop = chatMessages.scrollHeight;
        }
        
        function showSyncStatus(text) {
            syncStatus.textContent = `ðŸ”„ ${text}`;
            syncStatus.classList.add('active');
            setTimeout(() => {
                syncStatus.classList.remove('active');
            }, 3000);
        }
        
        function formatTime(seconds) {
            const mins = Math.floor(seconds / 60);
            const secs = Math.floor(seconds % 60);
            return `${mins}:${secs.toString().padStart(2, '0')}`;
        }
        
        // Enter key for video URL
        videoUrl.addEventListener('keypress', (e) => {
            if (e.key === 'Enter') loadBtn.click();
        });
        
        // Auto-focus chat input
        chatInput.focus();
    </script>
